class Solution {
    public int[] countBits(int n) {
        int[] array=new int[n+1];

        for(int i=0;i<array.length;i++){
            array[i]=hammingWeight(i);

        }


        return array;
        
    }


  public int hammingWeight(int n) {
        int count = 0;
        while (n != 0) {
            count += n & 1; // 檢查最低位是否為1
            n >>>= 1; // 將整數無符號右移一位
        }
        return count;
  }

  }
